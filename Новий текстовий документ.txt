LibraryDBcontext db = new LibraryDBcontext();
var books = from Book in db.Books
            join Author in db.Authors
            on Book.AuthorID equals Author.ID
            join Genre in db.Genres
            on Book.GenreID equals Genre.ID
            where Author.ID == Book.AuthorID && Genre.ID == Book.GenreID
            select new {
                Book.ID,
                Book.Name,
                Author.FirstName,
                Author.LastName,

                GenreName = Genre.Name
            };
foreach (var book in books)
{
    Console.WriteLine($"{book.ID}|{book.Name}|{book.LastName}|{book.FirstName}|{book.GenreName}");
}
